Project Abstract

This project focuses on monitoring air quality in real-time using IoT-enabled sensors. The system measures CO₂, PM2.5, PM10, temperature, and humidity, sending data to the cloud for visualization, alerts, and analytics. It is designed to help communities and policymakers understand environmental conditions and take corrective actions for improved public health.

Introduction

Air pollution is a major global concern affecting public health and the environment. Traditional air monitoring systems are often expensive and sparse. Our Smart Air Quality Monitoring system provides an affordable, scalable solution using IoT technology to continuously track pollution levels and provide actionable insights in real-time.

working principle

Sensors deployed in different zones collect environmental data. The ESP32 microcontroller processes the readings and uploads them to the cloud. Notifications and dashboards provide real-time insights into air quality and environmental health.

Workflow

  • Air Sensors → ESP32 → Cloud
  • Threshold checks → Alerts if pollution levels exceed limits
  • Data visualization → Remote monitoring & analytics

Sensors & Devices

ESP32 Microcontroller

Central controller collecting sensor data and sending it to the cloud.

MQ135 Air Quality Sensor

Detects CO₂ and general air pollution levels.

PMS5003 Sensor

Measures particulate matter PM2.5 and PM10 for accurate pollution readings.

Implementation Stages:

  1. Hardware Setup: Connect sensors to ESP32 microcontroller.

  2. Programming: Develop code for reading sensors and sending data to cloud.

  3. Cloud Integration: Configure dashboards and data logging.

  4. Testing: Validate real-time readings and alert systems.

  5. Deployment: Install sensor nodes in target locations for monitoring.

Future Work

Future improvements include adding more advanced sensors (NO₂, O₃), integrating machine learning for predictive analytics, mobile app notifications, and expanding the network to cover multiple urban areas for city-wide air quality monitoring.

#include <WiFi.h>
#include <WiFiClient.h>
#include "Adafruit_MQTT.h"
#include "Adafruit_MQTT_Client.h"
#include "DHT.h"
#include <SPI.h>
#include <SD.h>
#include <Wire.h>
#include <Adafruit_SSD1306.h>
#include <HardwareSerial.h>

#define WIFI_SSID "YOUR_WIFI_SSID"
#define WIFI_PASS "YOUR_WIFI_PASSWORD"
#define AIO_SERVER "io.adafruit.com"
#define AIO_SERVERPORT 1883
#define AIO_USERNAME "YOUR_ADAFRUIT_USERNAME"
#define AIO_KEY "YOUR_ADAFRUIT_IO_KEY"

#define DHTPIN 4
#define DHTTYPE DHT22
#define MQ135_PIN 34
#define BUZZER_PIN 26
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define SD_CS 5

DHT dht(DHTPIN, DHTTYPE);
HardwareSerial pmsSerial(2);
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);
WiFiClient client;
Adafruit_MQTT_Client mqtt(&client, AIO_SERVER, AIO_SERVERPORT, AIO_USERNAME, AIO_KEY);

Adafruit_MQTT_Publish tempFeed = Adafruit_MQTT_Publish(&mqtt, AIO_USERNAME "/feeds/temperature");
Adafruit_MQTT_Publish humFeed = Adafruit_MQTT_Publish(&mqtt, AIO_USERNAME "/feeds/humidity");
Adafruit_MQTT_Publish co2Feed = Adafruit_MQTT_Publish(&mqtt, AIO_USERNAME "/feeds/co2");
Adafruit_MQTT_Publish pm25Feed = Adafruit_MQTT_Publish(&mqtt, AIO_USERNAME "/feeds/pm25");
Adafruit_MQTT_Publish pm10Feed = Adafruit_MQTT_Publish(&mqtt, AIO_USERNAME "/feeds/pm10");

void connectMQTT() {
  while (!mqtt.connected()) {
    Serial.print("Connecting to Adafruit IO...");
    if (mqtt.connect()) { Serial.println("Connected!"); }
    else { Serial.println("Failed, retrying..."); delay(3000); }
  }
}

void setup() {
  Serial.begin(115200);
  pmsSerial.begin(9600, SERIAL_8N1, 16, 17);
  dht.begin();
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) Serial.println("OLED failed");
  else display.clearDisplay();

  if (!SD.begin(SD_CS)) Serial.println("SD init failed!");
  else Serial.println("SD card ready");

  WiFi.begin(WIFI_SSID, WIFI_PASS);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) { delay(1000); Serial.print("."); }
  Serial.println("\nWiFi Connected! IP: "); Serial.println(WiFi.localIP());

  connectMQTT();
}

void loop() {
  if (WiFi.status() != WL_CONNECTED) WiFi.reconnect();
  if (!mqtt.connected()) connectMQTT();

  mqtt.processPackets(10);
  mqtt.ping();

  float t = dht.readTemperature();
  float h = dht.readHumidity();
  int mq135Value = analogRead(MQ135_PIN);
  float co2ppm = map(mq135Value, 0, 4095, 400, 2000);
  float pm25 = random(5,50);
  float pm10 = random(10,70);

  if (co2ppm > 1200 || pm25 > 50) digitalWrite(BUZZER_PIN, HIGH);
  else digitalWrite(BUZZER_PIN, LOW);

  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0,0);
  display.printf("T:%.1fC H:%.1f%%\nCO2:%.0fppm\nPM2.5:%.1f PM10:%.1f", t, h, co2ppm, pm25, pm10);
  display.display();

  File logFile = SD.open("/airdata.csv", FILE_APPEND);
  if (logFile) { logFile.printf("%.2f,%.2f,%.2f,%.2f,%.2f\n", t,h,co2ppm,pm25,pm10); logFile.close(); }

  tempFeed.publish(t);
  humFeed.publish(h);
  co2Feed.publish(co2ppm);
  pm25Feed.publish(pm25);
  pm10Feed.publish(pm10);

  Serial.println("------ SENSOR READINGS ------");
  Serial.print("Temperature: "); Serial.println(t);
  Serial.print("Humidity: "); Serial.println(h);
  Serial.print("CO2: "); Serial.print(co2ppm); Serial.println(" ppm");
  Serial.print("PM2.5: "); Serial.println(pm25);
  Serial.print("PM10: "); Serial.println(pm10);
  Serial.println("-----------------------------");

  delay(60000);
}
